    <script>
        // Массив пожеланий
        const greetings = [

        const reasons = [
            "У тебя самая красивая улыбка на свете",
            "Твой смех звучит как самая лучшая музыка",
            "Ты делаешь этот мир лучше просто своим присутствием",
            "С тобой каждый день - настоящий праздник",
            "Ты веришь в меня даже тогда, когда я сам в себя не верю",
            "Ты готовишь самые вкусные завтраки",
            "Ты умеешь поддержать в трудную минуту",
            "Ты самая заботливая и нежная",
            "Ты всегда находишь слова, чтобы поднять настроение",
            "Ты просто - Ты, и это самое лучшее!"
        ];
        
        let reasonCount = 0;
        let musicPlaying = false;
        const audio = document.getElementById('bgMusic');
        
        function changeGreeting() {
            const messageElement = document.getElementById('greetingMessage');
            const randomIndex = Math.floor(Math.random() * greetings.length);
            messageElement.textContent = greetings[randomIndex];
            messageElement.style.animation = 'none';
            messageElement.offsetHeight;
            messageElement.style.animation = 'cardAppear 0.5s ease-out';
        }
        
        function showNextReason() {
            const reasonElement = document.getElementById('reasonDisplay');
            reasonElement.textContent = reasons[reasonCount % reasons.length];
            reasonCount++;
            reasonElement.style.animation = 'none';
            reasonElement.offsetHeight;
            reasonElement.style.animation = 'fadeIn 0.5s';
        }
        
        function changeTheme(color) {
            document.querySelector('.message').style.borderLeftColor = color;
            document.querySelectorAll('.button').forEach(btn => {
                if(!btn.classList.contains('surprise')) {
                    btn.style.background = `linear-gradient(45deg, ${color}, ${adjustBrightness(color, 20)})`;
                }
            });
        }
        
        function adjustBrightness(hex, percent) {
            // Простая функция для осветления цвета
            return hex; // Для простоты оставим так
        }
        
        function showSurprise() {
            const surpriseModal = document.createElement('div');
            surpriseModal.className = 'surprise-modal';
            surpriseModal.innerHTML = `
                <div class="surprise-content">
                    <h2>Ты - моё вдохновение! 💕</h2>
                    <div class="photo-frame">
                        <img src="https://via.placeholder.com/300x300/ff9a9e/ffffff?text=Тут+могла+быть+твоя+фотка" alt="Любимая">
                    </div>
                    <p>Самая красивая, самая любимая, самая родная!</p>
                    <p style="font-size: 2rem; margin: 20px 0;">💝 🌹 💝</p>
                    <button class="close-btn" onclick="this.parentElement.parentElement.remove()">Закрыть</button>
                </div>
            `;
            document.body.appendChild(surpriseModal);
            
            // Анимация появления
            setTimeout(() => {
                surpriseModal.style.opacity = '1';
            }, 10);
        }
        
        function toggleMusic() {
            const btn = document.querySelector('.music-btn');
            if (musicPlaying) {
                audio.pause();
                btn.textContent = '🎵 Включить музыку';
            } else {
                audio.play().catch(e => console.log('Автовоспроизведение заблокировано браузером'));
                btn.textContent = '⏸ Выключить музыку';
            }
            musicPlaying = !musicPlaying;
        }
        
        // Создание плавающих цветов
        function createFlowers() {
            const flowersContainer = document.getElementById('flowers');
            const flowers = ['🌸', '🌷', '🌹', '🌺', '🌻', '🌼', '💐'];
            
            for (let i = 0; i < 25; i++) {
                const flower = document.createElement('div');
                flower.className = 'flower';
                flower.textContent = flowers[Math.floor(Math.random() * flowers.length)];
                flower.style.left = Math.random() * 100 + '%';
                flower.style.top = Math.random() * 100 + '%';
                flower.style.fontSize = (Math.random() * 30 + 20) + 'px';
                flower.style.animationDelay = Math.random() * 5 + 's';
                flower.style.animationDuration = (Math.random() * 4 + 4) + 's';
                flowersContainer.appendChild(flower);
            }
        }
        
        // Создание сердечек
        function createHearts() {
            const heartsContainer = document.getElementById('hearts');
            
            setInterval(() => {
                const heart = document.createElement('div');
                heart.className = 'heart';
                heart.textContent = '❤️';
                heart.style.left = Math.random() * 100 + '%';
                heart.style.animationDuration = (Math.random() * 2 + 2) + 's';
                heart.style.fontSize = (Math.random() * 20 + 10) + 'px';
                heart.style.animationDelay = Math.random() * 2 + 's';
                heartsContainer.appendChild(heart);
                
                setTimeout(() => {
                    heart.remove();
                }, 4000);
            }, 200);
        }
        
        // Дата
        function updateDate() {
            const dateElement = document.getElementById('currentDate');
            const now = new Date();
            const options = { year: 'numeric', month: 'long', day: 'numeric' };
            dateElement.textContent = now.toLocaleDateString('ru-RU', options);
        }
